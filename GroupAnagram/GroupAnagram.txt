public class GroupAnagram{
	public static void main (String[] args){
		System.out.println(groupAnagrams(["kofi", "ama", "fiko", "maa"]);
	}

	public static List<List<String>> groupAnagrams(String[] strgs){
		if(strgs.length() == 0){
			return new ArrayList();
		}
		
		HashMap<String, List> ansMap = new HashMap<>();

		for(String word : strgs){
			char[] wordArrray = word.toCharArray();
			Arrays.sort(wordArray);
			String sortedWord = new String(wordArray);
			
			if(!ansMap.containsKey(sortedWord)){
				ansMap.put(sortedWord, new ArrayList<>());
			}
			ansMap.get(sortedWord).add(word);
		}
		return new ArrayList<>(ansMap.values());
		
	}
}
